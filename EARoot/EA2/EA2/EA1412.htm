4000, 1) DESCR_DISTR_MUNDIAL, <br/>(SELECT K.SCIENTIFICNAME  FROM TESP_TAXON K WHERE   K.TAXONID = T.IDKINGDOM) AS REINO,<br/>(SELECT P.SCIENTIFICNAME  FROM TESP_TAXON P WHERE   P.TAXONID = T.IDPHYLUM) AS FILO,<br/>(SELECT C.SCIENTIFICNAME  FROM TESP_TAXON C WHERE   C.TAXONID = T.IDCLASS) AS CLASE,<br/>(SELECT O.SCIENTIFICNAME  FROM TESP_TAXON O WHERE   O.TAXONID = T.IDORDER) AS ORDEN,<br/>(SELECT F.SCIENTIFICNAME  FROM TESP_TAXON F WHERE   F.TAXONID = T.IDFAMILY) AS FAMILIA,<br/>(SELECT G.SCIENTIFICNAME  FROM TESP_TAXON G WHERE   G.TAXONID = T.IDGENUS) AS GENERO,<br/>(SELECT S.SCIENTIFICNAME  FROM TESP_TAXON S WHERE   S.TAXONID = T.IDSUBGENUS) AS SUBGENRO,<br/>(SELECT W.SCIENTIFICNAME  FROM TESP_TAXON W WHERE   W.TAXONID = T.IDSPECIFICEPITHET) AS SPECIFICEPITHET,<br/>(SELECT I.SCIENTIFICNAME  FROM TESP_TAXON I WHERE   I.TAXONID = T.IDINFRASPECIFICEPITHET) AS INFRASPECIFICEPITHET,<br/>T.TAXONCONCEPTID, <br/>SYS.DBMS_LOB.SUBSTR(T.NAMEACCORDINGTO, 4000, 1) AS NAMEACCORDINGTO,<br/>SYS.DBMS_LOB.SUBSTR(T.NAMEPUBLISHEDIN, 4000, 1) AS NAMEPUBLISHEDIN,<br/>T.NAMEPUBLISHEDINYEAR,<br/>T.HIGHERCLASSIFICATION,<br/>K.TAXONRANK,<br/>T.SCIENTIFICNAMEAUTHORSHIP,<br/>SYS.DBMS_LOB.SUBSTR(T.TAXONREMARKS, 4000, 1) AS TAXONREMARKS,<br/>T.MODIFIED, T.EUNIS_CODE, T.CITES_CODE, T.REDNATURA_CODE, T.COD_AMBITO,<br/>TESP_COLECCIONES.COLECCION,<br/>SYS.DBMS_LOB.SUBSTR(M.MEASUREMENTVALUE, 4, 1) AS COD_HABITAT<br/><br/>FROM TESP_TAXON T <br/>  INNER JOIN TESP_TAXON_COLECCIONES  ON T.TaxonID = TESP_TAXON_COLECCIONES.TAXONID<br/>  INNER JOIN TESP_COLECCIONES ON TESP_TAXON_COLECCIONES.IDCOLECCION = TESP_COLECCIONES.IDCOLECCION<br/>  INNER JOIN TESP_T_TAXONRANK  K  ON K.IDTAXONRANK = T.IDTAXONRANK<br/>  INNER JOIN TESP_DWCMEASUREMENTORFACT  M  ON M.TAXONID = T.TAXONID<br/>  LEFT JOIN TESP_TAXON_TIENE_VERNACULAR R ON T.TAXONID = R.TAXONID <br/>  LEFT JOIN TESP_VERNACULARNAME V ON R.IDVERNACULARNAME = V.IDVERNACULARNAME<br/>WHERE  TESP_TAXON_COLECCIONES.IDCOLECCION = 15 AND T.IDGENUS IS NOT NULL  <br/>  AND M.IDMEASUREMENTMETHOD = 11 AND V.IDLANGUAGE = 1<br/>ORDER BY T.SCIENTIFICNAME;<br/><p/>
