<b>VISTA MATERIALIZADA: VM_LESPE</b><br/><br/>SELECT DISTINCT TESP_TAXON_COLECCIONES.IDCOLECCION, T.TAXONID, T.SCIENTIFICNAME, <br/>  V.VERNACULARNAME,<br/>  V.IDLANGUAGE,<br/>  T.HIGHERCLASSIFICATION,<br/>                (SELECT K.SCIENTIFICNAME  FROM TESP_TAXON K WHERE   K.TAXONID = T.IDKINGDOM) AS REINO,<br/>                (SELECT P.SCIENTIFICNAME  FROM TESP_TAXON P WHERE   P.TAXONID = T.IDPHYLUM) AS FILO,<br/>                (SELECT C.SCIENTIFICNAME  FROM TESP_TAXON C WHERE   C.TAXONID = T.IDCLASS) AS CLASE,<br/>                (SELECT O.SCIENTIFICNAME  FROM TESP_TAXON O WHERE   O.TAXONID = T.IDORDER) AS ORDEN,<br/>                (SELECT F.SCIENTIFICNAME  FROM TESP_TAXON F WHERE   F.TAXONID = T.IDFAMILY) AS FAMILIA,<br/>                (SELECT G.SCIENTIFICNAME  FROM TESP_TAXON G WHERE   G.TAXONID = T.IDGENUS) AS GENERO,<br/>                (SELECT S.SCIENTIFICNAME  FROM TESP_TAXON S WHERE   S.TAXONID = T.IDSUBGENUS) AS SUBGENRO,<br/>                (SELECT W.SCIENTIFICNAME  FROM TESP_TAXON W WHERE   W.TAXONID = T.IDSPECIFICEPITHET) AS SPECIFICEPITHET,<br/>                (SELECT I.SCIENTIFICNAME  FROM TESP_TAXON I WHERE   I.TAXONID = T.IDINFRASPECIFICEPITHET) AS INFRASPECIFICEPITHET,<br/>  TESP_T_AMBITOS.AMBITO<br/>FROM TESP_TAXON T <br/>  INNER JOIN TESP_TAXON_COLECCIONES ON T.TaxonID = TESP_TAXON_COLECCIONES.TAXONID<br/>  LEFT JOIN TESP_TAXON_TIENE_VERNACULAR R ON T.TAXONID = R.TAXONID <br/>  LEFT JOIN TESP_VERNACULARNAME V ON R.IDVERNACULARNAME = V.IDVERNACULARNAME<br/>  LEFT JOIN TESP_T_AMBITOS  ON TESP_T_AMBITOS.COD_AMBITO  = T.COD_AMBITO <br/>WHERE  TESP_TAXON_COLECCIONES.IDCOLECCION = 12 AND V.IDLANGUAGE = 1 AND T.IDGENUS IS NOT NULL  <br/>ORDER BY REINO, FILO, ORDEN, FAMILIA, GENERO, SUBGENRO, INFRASPECIFICEPITHET;<br/><p/>
